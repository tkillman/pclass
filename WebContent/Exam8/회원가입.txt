 public Vector zipcodeRead(String area3)  {

            con = getConnection();

                tempZipcode.setArea1(rs.getString("area1"));

            if (pstmt != null) try { pstmt.close(); } catch(SQLException ex) {}
            if (con != null) try { con.close(); } catch(SQLException ex) {}
        }
        return vecList;
